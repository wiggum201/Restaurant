CREATE TABLE investments(
id int NOT NULL,
user_id int NOT NULL,
payments_id float NOT NULL,
bank_accounts_id int NOT NULL,
real_estates_id int NOT NULL,
expenses_id float NOT NULL,
mortgages_id float NOT NULL,
PRIMARY KEY (id),
FOREIGN KEY (user_id) REFERENCES users(id),
FOREIGN KEY (payments_id) REFERENCES payments(id),
FOREIGN KEY (bank_accounts_id) REFERENCES bank_accounts (id)
FOREIGN KEY (real_estates_id) REFERENCES real_estates(id),
FOREIGN KEY (expenses_id) REFERENCES expenses(id),
FOREIGN KEY (mortgages_id) REFERENCES mortgages(id)
);

CREATE TABLE payments(
id int NOT NULL,
rent_mortgage float,
clock date,
amount float,
PRIMARY KEY (id)
);

CREATE TABLE bank_accounts(
id int NOT NULL,
balance float,
account_number int,
amount float,
PRIMARY KEY (id)
);

CREATE TABLE real_estates(
id int NOT NULL,
units int,
address varchar(255),
monthly_rent float,
PRIMARY KEY (id)
);

CREATE TABLE expenses(
id int NOT NULL,
clock date,
amount float,
description varchar(255),
PRIMARY KEY (id)
);

CREATE TABLE mortgages(
id int NOT NULL,
interest_rate float,
loan_number int,
next_due_date date,
payment_amount float,
original_balance float,
current_balance float,
PRIMARY KEY (id)
);
